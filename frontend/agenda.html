Você ainda não tem nenhum agendamento.

`; lucide.createIcons(); return; } lista.innerHTML = ''; agendamentos.forEach(ag => { // Formata a data (Ex: 25/10/2026 às 14:30) const dataObj = new Date(ag.data_hora); const dataFormatada = dataObj.toLocaleDateString('pt-BR') + ' às ' + dataObj.toLocaleTimeString('pt-BR', {hour: '2-digit', minute:'2-digit'}); // --- AJUSTE AQUI: Mudamos de 'usuarios' para 'cliente' --- const nomeCliente = ag.cliente ? ag.cliente.nome : 'Cliente Desconhecido'; const telefoneCliente = ag.cliente ? ag.cliente.telefone : ''; const foneLimpo = telefoneCliente ? telefoneCliente.replace(/\D/g, '') : ''; const linkZap = foneLimpo ? `https://wa.me/55${foneLimpo}?text=Olá ${encodeURIComponent(nomeCliente)}, sobre o seu agendamento no ConectaSul...` : '#'; // Cores para o status const statusCor = ag.status === 'pendente' ? 'bg-yellow-900/30 text-yellow-500 border-yellow-700/50' : 'bg-green-900/30 text-green-500 border-green-700/50'; lista.innerHTML += `
${dataFormatada} ${ag.status}

${nomeCliente}

Serviço: ${ag.descricao_servico}

${foneLimpo ? ` Chamar no Zap ` : `Sem telefone cadastrado`}
`; }); lucide.createIcons(); } catch (err) { console.error(err); lista.innerHTML = '
Erro ao carregar a agenda. Verifique o console.
'; } }
